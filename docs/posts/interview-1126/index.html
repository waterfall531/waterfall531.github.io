function fineComper($list,$target){
    foreach($list as $index => $item){
        $tmpIndex = $index;
        $nextTarget = $target - $item;

        //find

        unset($tmpList[$index]);
        $tmpIndex2 = array_search($nextTarget, $tmpList);

}
