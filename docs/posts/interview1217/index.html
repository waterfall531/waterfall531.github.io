演算法

一維陣列 n元素 轉n維陣列 並設置值

class Solution
{
    function toArray($list, $value = 'value')
    {
        $list = array_reverse($list);
        return $this->add($list, $value);
    }

    function add(&$list, $setValue = 'value')
    {
        if (empty($list)) {
            return $setValue;
        }

        $value          = array_pop($list);
        $output[$value] = $this->add($list, $setValue);
        return $output;
    }
}
